00081             const FontChar fc = mFont->getChar(mText[i]);

00084             bool visibleCharacter = fc.width  > 0;
00085             if (visibleCharacter) {
00086                 vec2 basePoint = caret + vec2{fc.xoffset, mFont->height()-fc.height-fc.yoffset};

00092                 mBounds.min = glm::min(mBounds.min, basePoint);
00093                 mBounds.max = glm::max(mBounds.max, (vec2)position[position.size()-2]);
00094 
00095                 vec2 scale{1.0f / mFont->scaleW(), 1.0f / mFont->scaleH()};
00096 
00097                 textureCoords.push_back(vec2{fc.x         , mFont->scaleH()-fc.y-fc.height}*scale);
00098                 textureCoords.push_back(vec2{fc.x+fc.width, mFont->scaleH()-fc.y-fc.height}*scale);
