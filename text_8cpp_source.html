00059     void Text::updateVertexBuffer() {
00060         if (!font || text.length() ==0) {
00061             meshData->setSubmesh(0, {}, MeshType::Triangles);
00062             mesh->setMeshData(meshData);
00063             return;

00065         vector<vec3> position;
00066         vector<vec2> textureCoords;
00067         vector<GLushort> indices;
00068         vec2 carent{0,0};
00069         bounds.max = vec2{-FLT_MAX};
00070         bounds.min = vec2{FLT_MAX};
00071         for (unsigned short i=0;i<text.length();i++){
00072             FontChar fc = font->getChar(text[i]);
00073             if (fc.width  == -1){
00074                 continue;
00075             }
00076             vec2 basePoint = carent+vec2{fc.xoffset,font->height()-fc.height-fc.yoffset};
00077 
00078             position.push_back(vec3{basePoint,0});
00079             position.push_back(vec3{basePoint+vec2{fc.width,0},0});
00080             position.push_back(vec3{basePoint+vec2{fc.width,fc.height},0});
00081             position.push_back(vec3{basePoint+vec2{0,fc.height},0});
00082 
00083             bounds.min = glm::min(bounds.min, basePoint);
00084             bounds.max = glm::max(bounds.max, (vec2)position[position.size()-2]);
00085 
00086             vec2 scale{1.0f/font->getScaleW(), 1.0f/font->getScaleH()};
00087 
00088             textureCoords.push_back(vec2{fc.x,font->getScaleH()-fc.y-fc.height}*scale);
00089             textureCoords.push_back(vec2{fc.x+fc.width,font->getScaleH()-fc.y-fc.height}*scale);
00090             textureCoords.push_back(vec2{fc.x+fc.width,font->getScaleH()-fc.y}*scale);
00091             textureCoords.push_back(vec2{fc.x,font->getScaleH()-fc.y}*scale);
00092 
00093             // push two triangles
00094             indices.push_back(i*4);
00095             indices.push_back(i*4+1);
00096             indices.push_back(i*4+2);
00097             indices.push_back(i*4+2);
00098             indices.push_back(i*4+3);
00099             indices.push_back(i*4);
00100 
00101             carent.x += fc.xadvance;
00102         }
00103         meshData->setPosition(position);
00104         meshData->setTexCoord0(textureCoords);
00105         meshData->setSubmesh(0,indices, MeshType::Triangles);
00106         mesh->setMeshData(meshData);
