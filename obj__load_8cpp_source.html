00148     ObjData loadObjData(std::string objSource, std::function<std::string (std::string)> materialMap){
00149         ObjData res;
00150         stringstream ss{objSource};
00151         const int bufferSize = 256;
00152         char buffer[bufferSize];
00153         char buffer2[bufferSize];
00154         bool initialComment = true;
00155         while (ss.good()){
00156             ss.getline(buffer, bufferSize);
00157             stringstream likeTokensizer{buffer};
00158             vector<string> tokens;
00159             while (likeTokensizer.good()) {
00160                 likeTokensizer.getline(buffer2, bufferSize, ' ');
00161                 tokens.push_back(buffer2);
00162             }
00163             if (tokens.size()==0){
00164                 continue;
00165             }
00166             initialComment = initialComment && (tokens[0] == "#");
00167             int currentIndex = static_cast<int>(res.faces.size()) + 1;
00168             if (tokens[0] == "#"){ // comment
00169                 if (initialComment){
00170                     res.initialComment += string(buffer + 2) + "\n";
00171                 }
00172             } else if (tokens[0] == "v"){ // vertex position
00173                 vec4 vertexPosition = toVec4(tokens);
00174                 res.vertexPositions.push_back(vertexPosition);
00175             } else if (tokens[0] == "vt"){ // vertex texture coordinates
00176                 vec3 textureCoord = toVec3(tokens);
00177                 res.textureCoords.push_back(textureCoord);
00178             } else if (tokens[0] == "vn"){ // vertex normal
00179                 vec3 normal = toVec3(tokens);
00180                 res.normals.push_back(normal);
00181             } else if (tokens[0] == "f"){ // face
00182                 res.faces.push_back(toObjFace(tokens));
00183             } else if (tokens[0] == "mtllib"){ // material library
00184                 string materialLib = materialMap(tokens[1]);
00185                 parseMaterialLib(materialLib, res);
00186             } else if (tokens[0] == "usemtl"){ // use material
00187                 res.materialChanges.push_back({currentIndex, tokens[1]});
00188             } else if (tokens[0] == "o"){ // named object
00189                 res.namedObjects.push_back({currentIndex, tokens[1]});
00190             } else if (tokens[0] == "g"){ // polygon group
00191                 res.polygonGroups.push_back({currentIndex, tokens[1]});
00192             } else if (tokens[0] == "s"){ // smoothing groups
00193                 int smoothingGroup = 0; // 0 = no smoothing
00194                 if (tokens[1] != "off"){
00195                     smoothingGroup = atoi(tokens[1].c_str());
00196                 }
00197                 res.smoothGroups.push_back({currentIndex, smoothingGroup});
00198             }
00199         }
00200 
00201         return res;
