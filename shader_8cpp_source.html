00288         if (shaderProgram){
00289             glDeleteShader(shaderProgram);
00290         }
00291         shaderProgram = glCreateProgram();
00292         vector<ShaderObj> shaderObjects;
00293         vector<ShaderType> requiredTypes{ShaderType::VertexShader, ShaderType::FragmentShader};
00294         for (auto & element : shaderSources) {
00295             auto shaderType = element.first;
00296             string & source = element.second;
00297             if (source.length()>0){
00298                 string precompiledSource = getPrecompiledSource(source);
00299                 
00300                 shaderObjects.push_back(compileShader(precompiledSource, shaderType));
00301                 // remove element from vector
00302                 requiredTypes.erase(std::remove(requiredTypes.begin(), requiredTypes.end(), shaderType), requiredTypes.end());
00303                 glAttachShader( shaderProgram,  shaderObjects.back());
00304             }
00305         }
00306         if (requiredTypes.size()>0){
00307             throw ShaderBuildException{"Both vertex and fragment shader required.", ShaderErrorType::IncompleteShader};
00308         }
00309         
00310         linkProgram();
00311         // shaderObjects deleted when goes out of scope (which is ok as long as program is not deleted)
00312         glUseProgram(shaderProgram);
00313 
00314         // clean up
00315         for (auto & shaderObject : shaderObjects){
00316             shaderObject.detach(shaderProgram);
00317         }
00318         shaderObjects.clear();
00319 
00320         shaderAttributes = getActiveShaderAttributes(shaderProgram);
00321         shaderUniforms = getActiveShaderUniforms(shaderProgram);
00322         
00323         shaderProgramChanged.notifyListeners(this);
