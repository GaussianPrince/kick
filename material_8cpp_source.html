00077             pos->second.value.mat4Value = value.value.mat4Value; // copy largest element
00078             if (pos->second.glType != value.glType){
00079                 pos->second.glType = value.glType;
00080                 updateShaderLocation(name, pos->second);
00081             }
00082         } else {
00083             // not found insert new
00084             auto insertedElement = currentMaterialData.emplace(make_pair(name, move(value)));
00085             bool didInsert = insertedElement.second;
00086             assert(didInsert);

00088             updateShaderLocation(name, insertedIterator->second);
