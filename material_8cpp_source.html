00090     void Material::setUniformData(std::string name, MaterialData&& value){
00091         auto pos = currentUniformData.find(name);

00093             pos->second.value.mat4Value = value.value.mat4Value; // copy largest element
00094             if (pos->second.glType != value.glType){
00095                 pos->second.glType = value.glType;

00202     void Material::setDefaultUniforms() {
00203         vector<string> unmappedOrDefaultUniforms;
00204         for (auto& u : mShader->getShaderUniforms()){
00205             bool isAutoMapped = u.name.length()>0 && u.name[0] == '_';
00206             if (isAutoMapped){
00207                 continue;
00208             }
00209             auto currentValue = currentUniformData.find(u.name);
00210             bool isUnmapped = currentValue == currentUniformData.end();
00211             if (isUnmapped || currentValue->second.defaultUniform){
00212                 unmappedOrDefaultUniforms.push_back(u.name);
00213             }
00214         }
00215         for (auto & name : unmappedOrDefaultUniforms){
00216             MaterialData mat{0};
00217             if (mShader->tryGetDefaultUniform(name, mat)){
00218                 setUniformData(name, move(mat));
00219             }
00220         }
