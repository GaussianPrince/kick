00218     void MeshData::recomputeNormals(){
00219         // empty
00220         normal.resize(position.size());
00221         for (int i=0;i<normal.size();i++){
00222             normal[i] = vec3{0};
00223         }
00224         
00225         for (auto & v : subMeshes){
00226             auto & submesh = v.indices;
00227             if (v.meshType != MeshType::Triangles){
00228                 throw std::invalid_argument("Recalcualte normals only valid for triangles");
00229             }
00230             size_t triangleCount = submesh.size()/3;
00231             for (int a = 0; a < triangleCount; a++) {
00232                 GLuint i1 = submesh[a * 3];
00233                 GLuint i2 = submesh[a * 3 + 1];
00234                 GLuint i3 = submesh[a * 3 + 2];
00235                 
00236                 vec3 v1 = position[i1];
00237                 vec3 v2 = position[i2];
00238                 vec3 v3 = position[i3];
00239                 
00240                 vec3 v1v2 = normalize(v2 - v1);
00241                 vec3 v1v3 = normalize(v3 - v1);
00242                 
00243                 vec3 faceNormal = normalize(cross(v1v2, v1v3));
00244                 float weight1 = acos(std::max(-1.0f, std::min(1.0f, dot(v1v2, v1v3))));
00245                 
00246                 vec3 v2v3 = normalize(v3 - v2);
00247                 float weight2 = M_PI - acos(std::max(-1.0f, std::min(1.0f, dot(v1v2, v2v3))));
00248                 float weight3 = M_PI - weight1 - weight2;
00249                 normal[i1] += weight1 * faceNormal;
00250                 normal[i2] += weight2 * faceNormal;
00251                 normal[i3] += weight3 * faceNormal;
00252             }
00253         }
00254         for (int i=0;i<normal.size();i++){
00255             if (normal[i] != vec3{0}){
00256                 normal[i] = normalize(normal[i]);
00257             }
00258         }
00259     }
