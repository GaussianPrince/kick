00092         SDL_Surface* imgRef = *image;
00093         SDL_PixelFormat * format = imgRef->format;
00094         auto pixelFormat = format->format;
00095         bool isBGR = format->Rshift == 16;
00096         GLenum RGB  = isBGR ? GL_BGR : GL_RGB;
00097         GLenum RGBA = isBGR ? GL_BGRA : GL_RGBA;
00098         const bool alpha = SDL_ISPIXELFORMAT_ALPHA(pixelFormat);
00099         if (alpha){
00100             imageFormat.format = RGBA;
00101             imageFormat.internalFormat = GL_RGBA;
00102             imageFormat.type = GL_UNSIGNED_BYTE;
00103         } else {
00104             if (format->BytesPerPixel == 4){
00105                 imageFormat.format = RGBA;
00106                 imageFormat.internalFormat = GL_RGBA;
00107             } else {
00108                 imageFormat.format = RGB;
00109                 imageFormat.internalFormat = GL_RGB;
00110             }
00111             imageFormat.type = GL_UNSIGNED_BYTE;
00112         }

00116     {
00117         auto temp_row = unique_ptr<char>(new char[pitch]);
00118         if (temp_row.get() == nullptr)
00119         {
00120             SDL_SetError("Not enough memory for image inversion");
00121             return -1;
00122         }
00123         //if height is odd, don't need to swap middle row
00124         int height_div_2 = (int) (height * .5f);
00125         for(int index = 0; index < height_div_2; index++)    {
00126             //uses string.h
00127             memcpy((Uint8 *)temp_row.get(),
00128                    (Uint8 *)(image_pixels) +
00129                    pitch * index,
00130                    pitch);
00131             
00132             memcpy(
00133                    (Uint8 *)(image_pixels) +
00134                    pitch * index,
00135                    (Uint8 *)(image_pixels) +
00136                    pitch * (height - index-1),
00137                    pitch);
00138             memcpy(
00139                    (Uint8 *)(image_pixels) +
00140                    pitch * (height - index-1),
00141                    temp_row.get(),
00142                    pitch);
00143         }
00144         return 0;
