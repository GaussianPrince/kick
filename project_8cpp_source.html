00102         SDL_Surface* imgRef = *image;
00103         SDL_PixelFormat * format = imgRef->format;
00104         auto pixelFormat = format->format;
00105         bool isBGR = format->Rshift == 16;
00106         GLenum RGB  = isBGR ? GL_BGR : GL_RGB;
00107         GLenum RGBA = isBGR ? GL_BGRA : GL_RGBA;
00108         const bool alpha = SDL_ISPIXELFORMAT_ALPHA(pixelFormat);
00109         if (alpha){
00110             imageFormat.format = RGBA;
00111             imageFormat.internalFormat = GL_RGBA;
00112             imageFormat.type = GL_UNSIGNED_BYTE;
00113         } else {
00114             if (format->BytesPerPixel == 4){
00115                 imageFormat.format = RGBA;
00116                 imageFormat.internalFormat = GL_RGBA;
00117             } else {
00118                 imageFormat.format = RGB;
00119                 imageFormat.internalFormat = GL_RGB;
00120             }
00121             imageFormat.type = GL_UNSIGNED_BYTE;

00126     {
00127         auto temp_row = unique_ptr<char>(new char[width]);

00129         {
00130             SDL_SetError("Not enough memory for image inversion");

00132         }
00133         //if height is odd, don't need to swap middle row
00134         int height_div_2 = (int) (height * .5f);
00135         for(int index = 0; index < height_div_2; index++)    {
00136             //uses string.h
00137             memcpy((Uint8 *)temp_row.get(),
00138                    (Uint8 *)(image_pixels) +
00139                    width * index,
00140                     width);
00141             
00142             memcpy(
00143                    (Uint8 *)(image_pixels) +
00144                    width * index,
00145                    (Uint8 *)(image_pixels) +
00146                    width * (height - index-1),
00147                     width);
00148             memcpy(
00149                    (Uint8 *)(image_pixels) +
00150                    width * (height - index-1),
00151                    temp_row.get(),
00152                     width);
00153         }
00154         return 0;
