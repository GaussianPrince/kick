00138     {
00139         auto temp_row = unique_ptr<char>(new char[width]);
00140         if (temp_row.get() == nullptr)
00141         {
00142             SDL_SetError("Not enough memory for image inversion");
00143             return -1;
00144         }
00145         //if height is odd, don't need to swap middle row
00146         int height_div_2 = (int) (height * .5f);
00147         for(int index = 0; index < height_div_2; index++)    {
00148             //uses string.h
00149             memcpy((Uint8 *)temp_row.get(),
00150                    (Uint8 *)(image_pixels) +
00151                    width * index,
00152                     width);
00153             
00154             memcpy(
00155                    (Uint8 *)(image_pixels) +
00156                    width * index,
00157                    (Uint8 *)(image_pixels) +
00158                    width * (height - index-1),
00159                     width);
00160             memcpy(
00161                    (Uint8 *)(image_pixels) +
00162                    width * (height - index-1),
00163                    temp_row.get(),
00164                     width);
00165         }
