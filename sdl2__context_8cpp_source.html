00053     bool SDL2Context::showWindow(const WindowConfig& config){
00054         const char *windowName = config.name.c_str();
00055         int width = config.width;
00056         int height = config.height;
00057         int depthSize = config.depthBufferSize;
00058         
00059         viewportDim = glm::vec2(width, height);
00060         
00061         /* Request opengl 3.2 context.
00062          * SDL doesn't have the ability to choose which profile at this time of writing,
00063          * but it should default to the core profile */
00064         SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, 3);
00065         SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, 2);
00066         SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, SDL_GL_CONTEXT_PROFILE_CORE);
00067         
00068         if (depthSize>0){
00069             /* Turn on double buffering with a 24bit Z buffer.
00070              * You may need to change this to 16 or 32 for your system */
00071             SDL_GL_SetAttribute(SDL_GL_DOUBLEBUFFER, 1);
00072             SDL_GL_SetAttribute(SDL_GL_DEPTH_SIZE, 24);
00073         }
00074         
00075         /* Create our window centered at 512x512 resolution */
00076         window = SDL_CreateWindow(windowName, SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED,
00077                                   width, height, SDL_WINDOW_OPENGL | SDL_WINDOW_SHOWN);
00078         if (!window)
00079             return false;
00080         
00081         context = SDL_GL_CreateContext(window);
00082         return true;
00083     }
