00066     bool SDL2Context::showWindow(const WindowConfig& config){
00067         const char *windowName = config.name.c_str();
00068         int width = config.width;
00069         int height = config.height;
00070         int depthSize = config.depthBufferSize;
00071         
00072         viewportDim = glm::vec2(width, height);
00073         
00074         /* Request opengl 3.2 context.
00075          * SDL doesn't have the ability to choose which profile at this time of writing,
00076          * but it should default to the core profile */
00077         SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, 3);
00078         SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, 2);
00079         SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, SDL_GL_CONTEXT_PROFILE_CORE);
00080         
00081         if (depthSize>0){
00082             /* Turn on double buffering with a 24bit Z buffer.
00083              * You may need to change this to 16 or 32 for your system */
00084             SDL_GL_SetAttribute(SDL_GL_DOUBLEBUFFER, 1);
00085             SDL_GL_SetAttribute(SDL_GL_DEPTH_SIZE, depthSize);
00086         }
00087         
00088         /* Create our window centered at 512x512 resolution */
00089         window = SDL_CreateWindow(windowName, SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED,
00090                                   width, height, SDL_WINDOW_OPENGL | SDL_WINDOW_SHOWN);
00091         if (!window)
00092             return false;
00093         
00094         context = SDL_GL_CreateContext(window);
00095         return true;
00096     }
