00097 #endif
00098 
00099         
00100         if (depthSize>0){
00101             /* Turn on double buffering with a 24bit Z buffer.
00102              * You may need to change this to 16 or 32 for your system */
00103             SDL_GL_SetAttribute(SDL_GL_DOUBLEBUFFER, 1);

00106         
00107         /* Create our window centered at 512x512 resolution */
