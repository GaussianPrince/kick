00110     MeshData * MeshFactory::createUVSphereData(unsigned short slices, unsigned short stacks, float radius){
00111         auto meshData = Project::createAsset<MeshData>();
00112         size_t vertexCount = (stacks+1) * (slices + 1);
00113         vector<vec3> vertices{vertexCount};
00114         vector<vec2> uvs{vertexCount};
00115         vector<vec3> normals{vertexCount};
00116         vector<vec4> colors{vertexCount};
00117         vector<GLushort> indices;
00118 
00119         int index = 0;
00120         // create vertices
00121         for (unsigned short j = 0; j <= stacks; j++) {
00122             float latitude1 = (M_PI / stacks) * j - (M_PI / 2);
00123             float sinLat1 = sin(latitude1);
00124             float cosLat1 = cos(latitude1);
00125             for (int i = 0; i <= slices; i++) {
00126                 float longitude = ((M_PI * 2) / slices) * i;
00127                 float sinLong = sin(longitude);
00128                 float cosLong = cos(longitude);
00129                 vec3 normal{cosLong * cosLat1,
00130                     sinLat1,
00131                     sinLong * cosLat1};
00132                 normal = normalize(normal);
00133                 normals[index] = normal;
00134                 uvs[index] = vec2{1 - i /(float) slices, j /(float) stacks};
00135                 vertices[index] = normal * radius;
00136                 colors[index] = vec4(normal*0.5f+vec3(0.5f), 1.0f);
00137                 index++;
00138             }
00139         }
00140         // create indices
00141         for (int j = 0; j < stacks; j++) {
00142             if (j > 0) {
00143                 indices.push_back(j * (slices + 1)); // make degenerate
00144             }
00145             for (int i = 0; i <= slices; i++) {
00146                 index = j * (slices + 1) + i;
00147                 indices.push_back(index);
00148                 indices.push_back(index + slices + 1);
00149             }
00150             if (j + 1 < stacks) {
00151                 indices.push_back(index + slices + 1); // make degenerate
00152             }
00153         }
00154         
00155         meshData->setPosition(vertices);
00156         meshData->setTexCoord0(uvs);
00157         meshData->setNormal(normals);
00158         meshData->setColor(colors);
00159         meshData->setSubmesh(0, indices, MeshType::TriangleStrip);
