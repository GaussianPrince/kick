00023     void SceneLights::recomputeLight(mat4 viewMatrix){
00024         mat3 viewMatrixRotation = (mat3)viewMatrix;
00025         if (directionalLight){
00026             // compute light direction
00027             vec3 lightDirection{0,0,-1};
00028             auto transform = directionalLight->getGameObject()->getTransform();
00029             directionalLightWorld = rotate(transform->getRotation(), lightDirection);
00030 
00031             // transform to eye space
00032             vec3 directionalLightDirection = normalize(viewMatrixRotation * directionalLightWorld);
00033             directionalLightData[0] = directionalLightDirection;
00034 
00035             // compute half vector
00036             directionalLightData[2] = -normalize(vec3(0,0,-1) + directionalLightDirection);
00037 
00038             // set color intensity
00039             directionalLightData[1] = directionalLight->getColorIntensity();
00040         }  else {
00041             directionalLightData = mat3(0);
00042         }
00043         int i = 0;
00044         for (;i<std::min((size_t)KICK_MAX_POINT_LIGHTS, pointLights.size());i++){
00045             Light * light = pointLights[i];
00046             Transform * transform = light->getGameObject()->getTransform();
00047 
00048             // save eyespace position
00049             pointLightData[i][0] = (vec3)(viewMatrix * vec4(transform->getPosition(),1.0f));
00050             pointLightData[i][1] = light->getColorIntensity();
00051             pointLightData[i][2] = light->getAttenuation();
00052         }
00053         // reset unused
00054         for (;i<KICK_MAX_POINT_LIGHTS;i++){
00055             pointLightData[i] = mat3(0);
00056         }
00057     }
