00092     void Scene::componentListener(Component* component, ComponentUpdateStatus status){
00093         componentEvents.notifyListeners({component, status});
00094         Camera *camera = dynamic_cast<Camera*>(component);
00095         Light *light = dynamic_cast<Light*>(component);
00096         if (status == ComponentUpdateStatus::Added){
00097             if (camera){
00098                 cameras.push_back(camera);
00099             } else if (light){
00100                 lights[light] = light->lightTypeChanged.createListener([&](Light* l){
00101                     rebuildSceneLights();
00102                 }, 0);
00103                 addLight(light);
00104             }
00105             if (component->isUpdateable()){
00106                 updatable.push_back(component);
00107             }
00108         }
00109         if (status == ComponentUpdateStatus::Removed){
00110             if (camera){
00111                 auto pos = find(cameras.begin(), cameras.end(), camera);
00112                 if (pos != cameras.end()){
00113                     cameras.erase(pos);
00114                 }
00115             } else if (light){
00116                 // rebuild lights
00117                 auto lightPos = lights.find(light);
00118                 if (lightPos != lights.end()){
00119                     lights.erase(lightPos);
00120                     rebuildSceneLights();
00121                 }
00122             }
00123             if (component->isUpdateable()){
00124                 auto pos = find(updatable.begin(), updatable.end(), component);
00125                 if (pos != updatable.end()){
00126                     updatable.erase(pos);
00127                 }
00128             }
00129         }
00130     }
